async function sendOfficialEval() { const n = document.getElementById('name').value; const p = document.getElementById('phone').value; const b = document.getElementById('branch').value; const key = localStorage.getItem('mawjat_api_key'); // تخزين المفتاح محلياً للأمان if(!p || !n || !key) return alert('يرجى إكمال البيانات والتأكد من مفتاح الوصول'); try { const res = await fetch('/api/send', { method: 'POST', headers: { 'Content-Type': 'application/json', 'x-api-key': key // إرسال المفتاح عبر Header }, body: JSON.stringify({phone: p, name: n, branch: b}) }); if(res.ok) { alert('✅ تم الإرسال بنجاح'); // تفريغ الحقول } else { alert('❌ فشل الإرسال: تأكد من صلاحية المفتاح'); } } catch (e) { alert('❌ خطأ في الاتصال بالسيرفر'); } }
